Future<void> addDefaultTemplates(String languageCode) async {
  if (box.isEmpty) {
    final defaultTemplates = languageCode == 'ar' 
        ? _getArabicTemplates() 
        : _getEnglishTemplates();
    
    for (var template in defaultTemplates) {
      await addTemplate(
        title: template['title']!,
        message: template['message']!,
      );
    }
  }
}
